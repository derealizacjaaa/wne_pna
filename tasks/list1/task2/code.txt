# Zadanie 2: Wprowadzenie wektorów i macierzy

# Wektor a: 7 elementów ln(10)
a <- rep(log(10), 7)

# Wektor b: ułamki
b <- c(1/3, 1/2, 1/6, 1/7)

# Wektor c: wartości z π
c <- c(pi, pi/3, pi/5, pi/7)

# Wektor d: liczby mieszane (dodatnie, ujemne, zera)
d <- c(-1, 1, 0, -2, 3, 0, 2)

# Wyświetlenie wyników
cat("Wektor a (ln 10 powtórzony 7 razy):\n")
print(a)

cat("\nWektor b (ułamki):\n")
print(b)

cat("\nWektor c (wartości z π):\n")
print(c)

cat("\nWektor d (liczby mieszane):\n")
print(d)
