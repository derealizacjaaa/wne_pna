<h2>Kod</h2>
<p>Tworzenie macierzy o specyficznej strukturze przy użyciu <code>diag()</code> i operacji na wierszach/kolumnach.</p>

code(
# Wymiar macierzy
n <- 11

# 1. Tworzymy macierz zerową
F_matrix <- matrix(0, nrow = n, ncol = n)

# 2. Pierwszy wiersz: same -1
F_matrix[1, ] <- -1

# 3. Tworzymy przesuniętą macierz jednostkową
# Podmacierz od wiersza 2 do n i od kolumny 1 do n-1
F_matrix[2:n, 1:(n-1)] <- diag(n-1)

cat("Macierz F (11x11):\n")
print(F_matrix)
)
