# Zadanie 2: Analiza meczów Anglia vs Szkocja

# Wczytanie danych z pliku CSV
dane <- read.csv("EngScoResults.csv", sep = ";")

cat("=== PODGLĄD DANYCH ===\n")
cat("Pierwsze wiersze:\n")
print(head(dane))
cat("\nWymiary danych:", nrow(dane), "wierszy,", ncol(dane), "kolumn\n\n")

# Podsumowanie danych
cat("Podsumowanie statystyczne:\n")
print(summary(dane))

# Ustawienie układu wykresów (1x2)
par(mfrow = c(1, 2))

# 1. Liczba goli Anglii w czasie
plot(dane$Rok, dane$Goals.England,
     main = "Gole Anglii w czasie",
     xlab = "Rok",
     ylab = "Liczba goli",
     pch = 19,
     col = "darkblue",
     cex = 0.8)
grid()
abline(lm(Goals.England ~ Rok, data = dane), col = "red", lwd = 2, lty = 2)

# 2. Gole Szkocji w meczach w Szkocji
mecze_szkocja <- dane[dane$Host == "Scotland", ]
plot(mecze_szkocja$Rok, mecze_szkocja$Goals.Scotland,
     main = "Gole Szkocji (mecze w Szkocji)",
     xlab = "Rok",
     ylab = "Liczba goli",
     pch = 19,
     col = "darkgreen",
     cex = 0.8)
grid()
abline(lm(Goals.Scotland ~ Rok, data = mecze_szkocja), col = "red", lwd = 2, lty = 2)

# Przywróć domyślny układ
par(mfrow = c(1, 1))

# Statystyki
cat("\n=== STATYSTYKI ===\n")
cat("Średnia goli Anglii:", mean(dane$Goals.England), "\n")
cat("Średnia goli Szkocji:", mean(dane$Goals.Scotland), "\n")
cat("Liczba meczów w Szkocji:", nrow(mecze_szkocja), "\n")
